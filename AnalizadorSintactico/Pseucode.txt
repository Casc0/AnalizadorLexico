
    class Token:
    def __init__(self, name, value, line_number):
        self.name = name          # Categoria: 'ID', 'NUM', 'IF', 'OP_REL', etc.
        self.value = value        # Atributo: puntero a tabla, valor especifico o vacio.
        self.line_number = line_number # Linea de origen en el codigo fuente.

    // --- Módulo Principal y Funciones Auxiliares ---

    MODULO principal() {
        // Inicializamos pidiendo el primer token a la clase LexiAnalyzer
        lookahead = lex.get_next_token();
        statement();
    }
    
    MODULO match(nombreEsperado) {
        // Evaluamos el atributo 'name' del objeto Token
        SI lookahead.name == nombreEsperado ENTONCES
            lookahead = lex.get_next_token();
        SINO
            reportarErrorSintactico("Se esperaba el token: " + nombreEsperado);
        FIN SI
    }
    
    // --- Procedimientos de la Gramática ---
    
    MODULO statement() {
        match('PROGRAM');
        ident();
        match('PUNTO_COMA');
        bloque();
        match('PUNTO');
    }
    
    MODULO bloque() {
        SI lookahead.name == 'VAR' ENTONCES
            match('VAR');
            sentencia_de_tipos();
            lista_subprogramas();
            cuerpo();
        SINO SI lookahead.name == 'FUNCTION' O lookahead.name == 'PROCEDURE' ENTONCES
            lista_subprogramas();
            cuerpo();
        SINO SI lookahead.name == 'BEGIN' ENTONCES
            cuerpo();
        SINO
            reportarErrorSintactico("Se esperaba VAR, FUNCTION, PROCEDURE o BEGIN");
        FIN SI
    }
    
    MODULO sentencia_de_tipos() {
        declaracion_de_var();
        match('PUNTO_COMA');
        sentencia_de_tipos_prima();
    }
    
    MODULO sentencia_de_tipos_prima() {
        // Si el siguiente token es un identificador, es una nueva declaración
        SI lookahead.name == 'ID' ENTONCES
            sentencia_de_tipos();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO declaracion_de_var() {
        lista_var();
        match('DOS_PUNTOS');
        tipo();
    }
    
    MODULO lista_var() {
        ident();
        match('COMA');
        lista_var_prima();
    }
    
    MODULO lista_var_prima() {
        SI lookahead.name == 'ID' ENTONCES
            lista_var();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO tipo() {
        SI lookahead.name == 'INTEGER' ENTONCES
            match('INTEGER');
        SINO SI lookahead.name == 'BOOLEAN' ENTONCES
            match('BOOLEAN');
        SINO
            reportarErrorSintactico("Tipo de dato inválido");
        FIN SI
    }
    
    MODULO lista_subprogramas() {
        subprograma();
        lista_subprogramas_prima();
    }
    
    MODULO lista_subprogramas_prima() {
        SI lookahead.name == 'PUNTO_COMA' ENTONCES
            match('PUNTO_COMA');
            lista_subprogramas();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO subprograma() {
        SI lookahead.name == 'FUNCTION' ENTONCES
            funcion();
        SINO SI lookahead.name == 'PROCEDURE' ENTONCES
            procedimiento();
        SINO
            reportarErrorSintactico("Se esperaba FUNCTION o PROCEDURE");
        FIN SI
    }
    
    MODULO funcion() {
        match('FUNCTION');
        ident();
        funcion_prima();
    }
    
    MODULO funcion_prima() {
        SI lookahead.name == 'PAR_ABRE' ENTONCES
            match('PAR_ABRE');
            param_formales();
            match('PAR_CIERRA');
            match('DOS_PUNTOS');
            tipo();
            match('PUNTO_COMA');
            bloque();
        SINO SI lookahead.name == 'DOS_PUNTOS' ENTONCES
            match('DOS_PUNTOS');
            tipo();
            match('PUNTO_COMA');
            bloque();
        SINO
            reportarErrorSintactico("Error en declaración de función");
        FIN SI
    }
    
    MODULO procedimiento() {
        match('PROCEDURE');
        ident();
        procedimiento_prima();
    }
    
    MODULO procedimiento_prima() {
        SI lookahead.name == 'PAR_ABRE' ENTONCES
            match('PAR_ABRE');
            param_formales();
            match('PAR_CIERRA');
            match('PUNTO_COMA');
            bloque();
        SINO SI lookahead.name == 'PUNTO_COMA' ENTONCES
            match('PUNTO_COMA');
            bloque();
        SINO
            reportarErrorSintactico("Error en declaración de procedimiento");
        FIN SI
    }
    
    MODULO param_formales() {
        lista_ident();
        match('DOS_PUNTOS');
        tipo();
        param_formales_prima();
    }
    
    MODULO param_formales_prima() {
        SI lookahead.name == 'PUNTO_COMA' ENTONCES
            match('PUNTO_COMA');
            param_formales();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO cuerpo() {
        match('BEGIN');
        lista_sentencias();
        match('END');
    }
    
    MODULO lista_sentencias() {
        sentencia();
        lista_sentencias_prima();
    }
    
    MODULO lista_sentencias_prima() {
        SI lookahead.name == 'PUNTO_COMA' ENTONCES
            match('PUNTO_COMA');
            lista_sentencias();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO sentencia() {
        SI lookahead.name == 'ID' ENTONCES
            ident();
            sentencia_prima();
        SINO SI lookahead.name == 'IF' ENTONCES
            alternativa();
        SINO SI lookahead.name == 'WHILE' ENTONCES
            repetitiva();
        SINO SI lookahead.name == 'BEGIN' ENTONCES
            cuerpo();
        SINO SI lookahead.name == 'READ' ENTONCES
            lectura();
        SINO SI lookahead.name == 'WRITE' ENTONCES
            escritura();
        SINO
            reportarErrorSintactico("Sentencia inválida");
        FIN SI
    }
    
    MODULO sentencia_prima() {
        SI lookahead.name == 'OP_ASIG' ENTONCES
            match('OP_ASIG');
            expresion();
        SINO SI lookahead.name == 'PAR_ABRE' ENTONCES
            match('PAR_ABRE');
            lista_expresion();
            match('PAR_CIERRA');
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO alternativa() {
        match('IF');
        expresion();
        match('THEN');
        sentencia();
        alternativa_prima();
    }
    
    MODULO alternativa_prima() {
        // Se fuerza a matchear el else con el if más cercano
        SI lookahead.name == 'ELSE' ENTONCES
            match('ELSE');
            sentencia();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO repetitiva() {
        match('WHILE');
        expresion();
        match('DO');
        sentencia();
    }
    
    MODULO lectura() {
        match('READ');
        match('PAR_ABRE');
        ident();
        match('PAR_CIERRA');
    }
    
    MODULO escritura() {
        match('WRITE');
        match('PAR_ABRE');
        expresion();
        match('PAR_CIERRA');
    }
    
    MODULO lista_expresion() {
        expresion();
        lista_expresion_prima();
    }
    
    MODULO lista_expresion_prima() {
        SI lookahead.name == 'COMA' ENTONCES
            match('COMA');
            lista_expresion();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO expresion() {
        expresion_simple();
        expresion_prima();
    }
    
    MODULO expresion_prima() {
        SI lookahead.name == 'OP_REL' ENTONCES
            relacion();
            expresion_simple();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO relacion() {
        SI lookahead.name == 'OP_REL' ENTONCES
            match('OP_REL');
        SINO
            reportarErrorSintactico("Se esperaba un operador relacional");
        FIN SI
    }
    
    MODULO expresion_simple() {
        SI lookahead.name == 'OP_ARIT' Y (lookahead.value == 'ADD' O lookahead.value == 'SUB') ENTONCES
            signo();
            termino();
            expresion_simple_prima();
        SINO SI es_inicio_de_factor() ENTONCES 
            termino();
            expresion_simple_prima();
        SINO
            reportarErrorSintactico("Error en expresión simple");
        FIN SI
    }
    
    MODULO expresion_simple_prima() {
        SI lookahead.name == 'OR' O (lookahead.name == 'OP_ARIT' Y (lookahead.value == 'ADD' O lookahead.value == 'SUB')) ENTONCES
            lista_expresion_simple();
            expresion_simple_prima();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO lista_expresion_simple() {
        SI lookahead.name == 'OR' ENTONCES
            match('OR');
            termino();
        SINO SI lookahead.name == 'OP_ARIT' Y (lookahead.value == 'ADD' O lookahead.value == 'SUB') ENTONCES
            signo();
            termino();
        SINO
            reportarErrorSintactico("Error en operador de expresión");
        FIN SI
    }
    
    MODULO signo() {
        SI lookahead.name == 'OP_ARIT' Y lookahead.value == 'ADD' ENTONCES
            match('OP_ARIT');
        SINO SI lookahead.name == 'OP_ARIT' Y lookahead.value == 'SUB' ENTONCES
            match('OP_ARIT');
        SINO
            reportarErrorSintactico("Se esperaba un signo (+ o -)");
        FIN SI
    }
    
    MODULO termino() {
        factor();
        termino_prima();
    }
    
    MODULO termino_prima() {
        SI lookahead.name == 'AND' O (lookahead.name == 'OP_ARIT' Y (lookahead.value == 'MUL' O lookahead.value == 'DIV')) ENTONCES
            lista_terminos();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO lista_terminos() {
        operacion();
        factor();
        lista_terminos_prima();
    }
    
    MODULO lista_terminos_prima() {
        SI lookahead.name == 'AND' O (lookahead.name == 'OP_ARIT' Y (lookahead.value == 'MUL' O lookahead.value == 'DIV')) ENTONCES
            lista_terminos();
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    MODULO operacion() {
        SI lookahead.name == 'OP_ARIT' Y lookahead.value == 'MUL' ENTONCES
            match('OP_ARIT');
        SINO SI lookahead.name == 'OP_ARIT' Y lookahead.value == 'DIV' ENTONCES
            match('OP_ARIT');
        SINO SI lookahead.name == 'AND' ENTONCES
            match('AND');
        SINO
            reportarErrorSintactico("Se esperaba '*', '/' o 'and'");
        FIN SI
    }
    
    MODULO factor() {
        SI lookahead.name == 'ID' ENTONCES
            ident();
            llamada_funcion();
        SINO SI lookahead.name == 'NUM' ENTONCES
            numero();
        SINO SI lookahead.name == 'TRUE' ENTONCES
            match('TRUE');
        SINO SI lookahead.name == 'FALSE' ENTONCES
            match('FALSE');
        SINO SI lookahead.name == 'PAR_ABRE' ENTONCES
            match('PAR_ABRE');
            expresion();
            match('PAR_CIERRA');
        SINO SI lookahead.name == 'NOT' ENTONCES
            match('NOT');
            factor();
        SINO
            reportarErrorSintactico("Error en factor");
        FIN SI
    }
    
    MODULO llamada_funcion() {
        SI lookahead.name == 'PAR_ABRE' ENTONCES
            match('PAR_ABRE');
            lista_expresion();
            match('PAR_CIERRA');
        // SINO: derivación en λ (no hace nada)
        FIN SI
    }
    
    // --- Integración con el Analizador Léxico (Laboratorio 2 y 3) ---
    
    MODULO ident() {
        match('ID');
    }
    
    MODULO numero() {
        match('NUM');
    }
    
    FUNCION es_inicio_de_factor() {
        RETORNAR (lookahead.name == 'ID' O lookahead.name == 'NUM' O lookahead.name == 'TRUE' O lookahead.name == 'FALSE' O lookahead.name == 'PAR_ABRE' O lookahead.name == 'NOT');
    }
